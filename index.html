// ===== 共通 ===== const $ = id => document.getElementById(id); // ===== 状態管理 ===== let questions = []; let index = 0; let score = 0; let total = 0; let bestScore = 0; let weakData = {}; let historyData = []; let reviewMode = false; let weakMode = false; let weakQuestions = []; let reviewIndex = 0; let wrongQuestions = []; let sessionStart = null; let sessionFinished = false; function loadCSV() { questions = []; fetch($("csvSelect").value) .then(r => { if (!r.ok) { $("question").textContent = "CSV読込失敗"; return ""; } return r.text(); }) .then(text => { questions = text .split("\n") .map(v => v.trim()) .filter(v => v) .map(line => { const p = line.split(","); return { question: p[0].trim(), answer: p[1].trim(), image: p[2] ? p[2].trim() : "" }; }); if (!questions.length) { $("question").textContent = "問題0件"; return; } sessionStart = new Date(); showQuestion(); updateScore(); }) .catch(() => { $("question").textContent = "CSV読込失敗"; }); } function showQuestion() { const q = questions[index]; $("question").textContent = q.question; const img = $("questionImage"); if (q.image) { img.src = "images/" + q.image; img.style.display = "block"; } else { img.style.display = "none"; } $("answer").value = ""; $("result").textContent = ""; } function updateScore() { const rate = total ? Math.round(score / total * 100) : 0; $("score").textContent = `${index + 1} / ${questions.length} 正解率: ${rate}%`; } function checkAnswer() { if (sessionFinished) return; const user = $("answer").value.trim().toLowerCase(); total++; const answers = questions[index] .answer .split("|") .map(v => v.trim().toLowerCase()); const correct = answers.includes(user); if (correct) { score++; $("result").textContent = "⭕ 正解"; } else { $("result").textContent = "❌ 正解: " + questions[index].answer; wrongQuestions.push(index); const q = questions[index].question; weakData[q] = (weakData[q] || 0) + 1; updateWeakRanking(); localStorage.setItem( "weakData", JSON.stringify(weakData) ); } updateScore(); } function nextQuestion() { if (sessionFinished) return; if (weakMode) { const pos = weakQuestions.indexOf(index) + 1; if (pos >= weakQuestions.length) { alert("苦手問題特訓完了"); weakMode = false; return; } index = weakQuestions[pos]; } else if (reviewMode) { reviewIndex++; if (reviewIndex >= wrongQuestions.length) { alert("ミス問題復習完了"); reviewMode = false; return; } index = wrongQuestions[reviewIndex]; } else { index++; if (index >= questions.length) { finishSession(); return; } } showQuestion(); } function startReview() { if (!wrongQuestions.length) { alert("ミス問題がありません"); return; } reviewMode = true; reviewIndex = 0; wrongQuestions = [...new Set(wrongQuestions)]; index = wrongQuestions[0]; sessionFinished = false; $("summary").innerHTML = "ミス問題復習中"; showQuestion(); } function startWeakMode() { weakQuestions = questions .map((q, i) => weakData[q.question] ? i : -1) .filter(i => i >= 0); if (!weakQuestions.length) { alert("苦手問題なし"); return; } weakMode = true; index = weakQuestions[0]; showQuestion(); } function updateWeakRanking() { const list = Object.entries(weakData) .map(([q, count]) => ({ q, count })) .sort((a, b) => b.count - a.count) .slice(0, 10); $("weak").innerHTML = list .map((v, i) => `${i + 1}位 ${v.q} (${v.count}回)` ) .join("
"); } function saveSessionHistory() { const d = new Date(); const stamp = `${d.getFullYear()}/${d.getMonth()+1}/${d.getDate()} ` + `${d.getHours()}:${("0"+d.getMinutes()).slice(-2)}`; const rate = Math.round(score / total * 100); const subject = $("csvSelect").options[ $("csvSelect").selectedIndex ].text; historyData.unshift( `${stamp} | ${subject} | ${score}/${total} | ${rate}%` ); historyData = historyData.slice(0,20); localStorage.setItem( "historyData", JSON.stringify(historyData) ); $("history").innerHTML = historyData.join("
"); } function finishSession() { sessionFinished = true; const end = new Date(); const minutes = Math.floor((end - sessionStart)/60000); const rate = Math.round(score / total * 100); const subject = $("csvSelect").options[ $("csvSelect").selectedIndex ].text; let html = `学習完了

科目：${subject}
正解数：${score}/${total}
正答率：${rate}%
学習時間：${minutes}分`; if(wrongQuestions.length){ html += `

ミス問題：${wrongQuestions.length}問


↓ ミス問題復習ボタンで再挑戦できます`; } $("summary").innerHTML = html; saveSessionHistory(); }
